import tkinter as tk
from tkinter import ttk

def geraMarca():
    
    marca = None

    def obterValorSelecionado():
        
        nonlocal marca
        
        valor = inputMarca.get()

        if valor == "AUDEN":
            
            marca = valor.lower()
            
        else:
            
            marca = f"portal{valor.lower()}"

        janela.destroy()

    janela = tk.Tk()
    janela.title("Bot de Renomear Turmas")

    label = tk.Label(janela, width=45, text="Selecione a Marca:")
    label.pack(pady=5)

    inputMarca = ttk.Combobox(janela, width=25, state="readonly")
    inputMarca["values"] = ["UNIBTA", "UFBRA"]
    inputMarca.pack(padx=10, pady=10)

    botao = tk.Button(janela, text="Selecionar Marca", command=obterValorSelecionado)
    botao.pack(pady=5)

    janela.mainloop()

    return marca
